00053     // holds 4-sided gutter dimensions (for margins and padding) .. no-export, header-only.
00054     struct Gutter
00055     {
00056         Gutter()
00057             : _top(0), _right(0), _bottom(0), _left(0) { }
00058         Gutter( float top, float right, float bottom, float left )
00059             : _top(top), _right(right), _bottom(bottom), _left(left) { }
00060         Gutter( float y, float x )
00061             : _top(y), _right(x), _bottom(y), _left(x) { }
00062         Gutter( float all )
00063             : _top(all), _right(all), _bottom(all), _left(all) { }
00064         bool operator !=( const Gutter& rhs ) const {
00065             return top() != rhs.top() || right() != rhs.right() || bottom() != rhs.bottom() || left() != rhs.left(); }
00066 
00067         float top()  const { return _top; }
00068         float left() const { return _left; }
00069         float right() const { return _right; }
00070         float bottom() const { return _bottom; }
00071 
00072         float x() const { return _left + _right; }
00073         float y() const { return _top + _bottom; }
00074 
00075         osg::Vec2f size() const { return osg::Vec2f(x(), y()); }
00076 
00077         osg::Vec2f offset() const { return osg::Vec2f( _left, _top ); }
00078 
00079     private:
00080         float _top, _right, _bottom, _left;
00081     };
