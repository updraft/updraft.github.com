00077 {
00078     WFSCapabilities *caps = NULL;
00079     if ( osgDB::containsServerAddress( location ) )
00080     {
00081         HTTPResponse response = HTTPClient::get( location, options );
00082         if ( response.isOK() && response.getNumParts() > 0 )
00083         {
00084             caps = read( response.getPartStream( 0 ) );
00085         }
00086     }
00087     else
00088     {
00089         if ((osgDB::fileExists(location)) && (osgDB::fileType(location) == osgDB::REGULAR_FILE))
00090         {
00091             std::ifstream in( location.c_str() );
00092             caps = read( in );
00093         }
00094     }
00095     return caps;
